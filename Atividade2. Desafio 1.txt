#include <stdio.h>
#include <stdlib.h>

typedef struct No {
    char c;
    struct No *prox;
} No;

typedef struct {
    No *topo;
} Pilha;

void init(Pilha *p) {
    p->topo = NULL;
}

int vazia(Pilha *p) {
    return p->topo == NULL;
}

void push(Pilha *p, char c) {
    No *n = (No*) malloc(sizeof(No));
    if(n != NULL) {
        n->c = c;
        n->prox = p->topo;
        p->topo = n;
    }
}

char pop(Pilha *p) {
    if(vazia(p)) return '\0';

    No *t = p->topo;
    char c = t->c;
    p->topo = t->prox;
    free(t);

    return c;
}

int par(char a, char f) {
    if(a == '(' && f == ')') return 1;
    if(a == '{' && f == '}') return 1;
    if(a == '[' && f == ']') return 1;
    return 0;
}

int main() {
    Pilha p;
    init(&p);

    char exp[100];
    int ok = 1;

    printf("Digite a expressao: ");
    scanf(" %[^\n]", exp);

    char *ptr = exp;

    while(*ptr != '\0') {
        if(*ptr == '(' || *ptr == '{' || *ptr == '[') {
            push(&p, *ptr);
        }
        else if(*ptr == ')' || *ptr == '}' || *ptr == ']') {
            if(vazia(&p)) {
                ok = 0;
                break;
            }

            char x = pop(&p);
            if(!par(x, *ptr)) {
                ok = 0;
                break;
            }
        }
        ptr++;
    }

    if(!vazia(&p)) ok = 0;

    if(ok)
        printf("Valida\n");
    else
        printf("Invalida\n");

    return 0;
}
